## Install

```bash
npm install confetti

yarn add confetti
```

## Usage

```tsx
import React from "react";
import Confetti from "confetti";

const App = () => {
  return (
    <div>
      <Confetti text="Modern React component module" />
    </div>
  );
};

export default App;
```
